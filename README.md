# About
Eye Component follows mouse


# Usage
```shell
#NPM
npm i react-eye

#Yarn
yarn add react-eye
```
### then in your React file
```js
import Eye from "react-eye";

const Sample:React.FC<Props> => (
  <>
    <Eye
      white={{
        x: 150,
        y: 150,
      }}
      iris={{
        x: 100,
        y: 100,
        color: "chocolate",
      }}
    />
  </>
)
```
